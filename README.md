"""PiezoPower Analyzer Streamlit app.

Run with:
    streamlit run app.py
"""

from __future__ import annotations

from pathlib import Path
from tempfile import NamedTemporaryFile

import matplotlib.pyplot as plt
import pandas as pd
import streamlit as st

from src.analyzer import AnalysisConfig, detect_steps, experiment_quality_score, normalize_columns, summarize_run
from src.report_generator import generate_pdf_report


ROOT = Path(__file__).parent
SAMPLE_DATA = ROOT / "sample_data" / "test_run_1.csv"


st.set_page_config(
    page_title="PiezoPower Analyzer",
    page_icon="⚡",
    layout="wide",
)

st.title("⚡ PiezoPower Analyzer")
st.caption("Python dashboard for analyzing voltage generated by piezoelectric floor tiles.")

with st.sidebar:
    st.header("Experiment settings")
    threshold = st.slider("Step detection threshold (V)", 0.1, 5.0, 1.2, 0.1)
    min_gap = st.slider("Minimum gap between steps (seconds)", 0.1, 1.5, 0.35, 0.05)
    capacitance_mf = st.number_input("Capacitance (mF)", min_value=0.01, max_value=100.0, value=1.0, step=0.1)
    st.info("Energy formula: E = 1/2 × C × V²")

uploaded_file = st.file_uploader("Upload Arduino CSV data", type=["csv"])

if uploaded_file is None:
    st.write("Using included demo dataset. Upload your own CSV when you have real Arduino data.")
    raw_data = pd.read_csv(SAMPLE_DATA)
else:
    raw_data = pd.read_csv(uploaded_file)

config = AnalysisConfig(
    threshold_volts=threshold,
    min_gap_seconds=min_gap,
    capacitance_farads=capacitance_mf / 1000,
)

try:
    data = normalize_columns(raw_data)
    steps = detect_steps(data, config)
    summary = summarize_run(data, steps, config)
    quality_score, quality_label = experiment_quality_score(summary)
except ValueError as exc:
    st.error(str(exc))
    st.stop()

metric_cols = st.columns(5)
metric_cols[0].metric("Steps detected", f"{int(summary['steps_detected'])}")
metric_cols[1].metric("Peak voltage", f"{summary['peak_step_voltage_v']:.2f} V")
metric_cols[2].metric("Avg voltage", f"{summary['average_voltage_v']:.2f} V")
metric_cols[3].metric("Energy estimate", f"{summary['total_energy_mj']:.3f} mJ")
metric_cols[4].metric("Quality score", f"{quality_score}/100", quality_label)

st.subheader("Voltage signal")
fig, ax = plt.subplots(figsize=(12, 5))
ax.plot(data["time_s"], data["voltage_v"], linewidth=1.2, label="Voltage")
ax.axhline(config.threshold_volts, linestyle="--", linewidth=1, label="Detection threshold")
if not steps.empty:
    ax.scatter(steps["time_s"], steps["peak_voltage_v"], marker="o", label="Detected steps")
ax.set_xlabel("Time (seconds)")
ax.set_ylabel("Voltage (V)")
ax.set_title("Piezoelectric Tile Voltage Over Time")
ax.legend()
ax.grid(True, alpha=0.25)
st.pyplot(fig)

left, right = st.columns([1.2, 1])
with left:
    st.subheader("Detected step peaks")
    if steps.empty:
        st.warning("No steps detected. Lower the threshold or check your CSV data.")
    else:
        display_steps = steps.copy()
        display_steps["energy_mj"] = display_steps["energy_j"] * 1000
        st.dataframe(
            display_steps[["step_number", "time_s", "peak_voltage_v", "energy_mj"]],
            use_container_width=True,
        )

with right:
    st.subheader("Experiment summary")
    summary_df = pd.DataFrame(
        [
            ["Duration", f"{summary['duration_s']:.2f} s"],
            ["Samples", f"{int(summary['samples'])}"],
            ["Step rate", f"{summary['steps_per_minute']:.2f} steps/min"],
            ["Maximum voltage", f"{summary['max_voltage_v']:.2f} V"],
            ["Total energy", f"{summary['total_energy_j']:.6f} J"],
        ],
        columns=["Metric", "Value"],
    )
    st.table(summary_df)

st.subheader("Download results")

results_csv = steps.to_csv(index=False).encode("utf-8")
st.download_button(
    label="Download detected steps CSV",
    data=results_csv,
    file_name="piezo_detected_steps.csv",
    mime="text/csv",
)

with NamedTemporaryFile(delete=False, suffix=".png") as chart_file:
    fig.savefig(chart_file.name, dpi=160, bbox_inches="tight")
    chart_path = Path(chart_file.name)

with NamedTemporaryFile(delete=False, suffix=".pdf") as pdf_file:
    pdf_path = Path(pdf_file.name)

generate_pdf_report(pdf_path, summary, steps, chart_path=chart_path)
st.download_button(
    label="Download PDF experiment report",
    data=pdf_path.read_bytes(),
    file_name="piezo_experiment_report.pdf",
    mime="application/pdf",
)

st.caption(
    "Note: Demo data is synthetic. Replace it with Arduino readings from your piezoelectric tile circuit. "
    "For real experiments, calibrate the ADC and account for rectifier losses and capacitor leakage."
)
